show databases;
use nationalzoo;
show tables;

create table country (
	country_id bigint PRIMARY KEY auto_increment,
    country_name varchar(255) NOT NULL
);

desc country;

create table state(
	state_id bigint PRIMARY KEY auto_increment,
    state_name varchar(255) not null,
    country_id bigint,
    foreign key (country_id) references Country(country_id) on delete cascade
);

desc state;

create table city(
	city_id bigint PRIMARY KEY auto_increment,
    city_name varchar(255) not null,
    state_id bigint,
    foreign key (state_id) references State(state_id) on delete cascade
);

desc city;

create table Address (
	address_id bigint PRIMARY KEY auto_increment,
    street varchar(255),
    zip_code varchar(255),
    country_id bigint,
    state_id bigint,
    city_id bigint,
    foreign key (country_id) references Country(country_id) on delete set Null,
    foreign key (state_id) references State(state_id) on delete set Null,
    foreign key (city_id) references City(city_id) on delete set Null
);



desc address;
    

create table user (
	user_id bigint PRIMARY KEY auto_increment,
    first_name varchar(255) not null,
    last_name varchar(255) not null,
    username varchar(255) not null,
    password varchar(255) not null,
    address_id bigint,
    foreign key (address_id) references Address(address_id) on delete set null
);

desc user;

create table zoo(
zoo_id bigint PRIMARY KEY auto_increment,
zoo_name varchar(255) not null,
address_id bigint,
foreign key (address_id) references Address(address_id) on delete set null
);

desc zoo;

create table Animal(
animal_id bigint PRIMARY KEY auto_increment,
animal_name varchar(255) not null,
animal_type varchar(255) not null,
zoo_id bigint,
user_id bigint,
foreign key (zoo_id) references Zoo(zoo_id) on delete cascade,
foreign key (user_id) references user(user_id) on delete set null
);

desc animal;
